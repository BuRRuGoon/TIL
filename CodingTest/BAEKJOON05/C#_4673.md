# 예제 함수 문제

## 내가 푼 방식
``` cs
int d(int num)
{
   int sum = num;
   while(num != 0)
   {
       sum = sum + (num % 10);
       num = num / 10;
   }
   return sum;
}

bool[] check = new bool[10001];

for(int i = 1; i < 10001; i++)
{
    int n = d(i);

    if (n < 10001) 
    {
        check[n] = true;
    }
}

for(int i = 1; i < 10001; i++)
{
    if(!check[i])
    {
        Console.WriteLine(i);
    }
}
```
