# 예제 그래프 문제 (푸는중)

## 내가 푼 방식
``` cs
int[] nmr = Array.ConvertAll(Console.ReadLine().Split(), int.Parse);
int n = nmr[0];
int m = nmr[1];
int r = nmr[2];
int[] visited = new int[n + 1];
List<int>[] graph = new List<int>[m + 1];
int count = 1;

// for(int i = 0; i < n + 1; i++){
//     graph.Append(new List<int>());
// }

for(int i = 0; i < m; i++)
{
    int[] uv = Array.ConvertAll(Console.ReadLine().Split(), int.Parse);
    int u = uv[0];
    int v = uv[1];

    if(graph[u] == null)
    {
        graph[u] = new List<int>();
    }
    graph[u].Append(v);

    if(graph[v] == null)
    {
        graph[v] = new List<int>();
    }
    graph[v].Append(u);
}

for (int i = 0; i < m; i++)
{
    if (graph[i]==null)
    {
        continue;
    }
    graph[i].Sort((x, y) => { return x.CompareTo(y); });
}

void dfs(int x)
{
    visited[x] = count;

    for(int i = 0; i < 2; i++)
    {
        int y = graph[x][i];
        Console.WriteLine(y);
        if(visited[y] == 0)
        {
            count += 1;
            dfs(y);
        }
    }
}


Console.WriteLine();
dfs(r);

Console.Write(String.Join("\n", visited.Skip(1)));
```
